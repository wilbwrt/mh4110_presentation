\section{Black-Scholes Equation}

\subsection{What is an Option?}

\begin{frame}{What is an Option?}
    \begin{itemize}[<+-| alert@+>] % Besides alert, use \pause also works
        \item Derivatives are financial contracts which derive value from underlying asset 
        \item Futures, Forwards, Swaps, Options
        \item Options:
        \begin{itemize}
            \item Gives the right, not the obligation, to buy/sell asset at specified price on specified date 
            \item Call gives right to buy
            \item Put gives right to sell
            \item American style can be utilised at any time up until expiration date
            \item European style can only be utilised on expiration date
        \end{itemize}
    \end{itemize}
\end{frame}

\subsection{Black-Scholes Background}

\begin{frame}{Black-Scholes Background}
    \begin{itemize}
        \item Developed in 1970s by Fischer Black, Myron Scholes, and Robert Merton 
        \item Provides an analytic solution to a second order parabolic partial differential equation by a transformation to the heat equation
        \item Leads to conclusion that options pricing dynamic is similar to heat dissipation
        \item Based on European style options
        \item American style options pricing can only be done numerically, not analytically
    \end{itemize}
\end{frame}

\subsection{Black-Scholes Equation}

\begin{frame}{Black-Scholes Equation}
    \begin{itemize}
        \item Second order parabolic partial differential equation
        \item For a stock which does not pay dividends, we have: 
    \end{itemize}
    \[
    \frac{\partial c}{partial t}
    + r S \frac{\partial c}{\partial S}
    + \frac{1}{2} \sigma^2 S^2 \frac{\partial^2 c}{\partial^2 S}
    = r c
    \]
    \begin{itemize}
        \item $\t$ = time
        \item $\c$ = call option price
        \item $\p$ = put option price
        \item $\S$ = spot stock price
        \item $\r$ = risk free interest rate 
        \item $\sigma$ = stock volatility
        \item $\S_0$ is spot stock price at time zero
        \item $\K$ is strike price
        \item $\T$ is time to maturity
        \item $\N(x)$ is Cumulative Distribution Probability
        \item Solutions to this partial differential equation are the Black-Scholes-Merton formulae for European call and put options prices
    \end{itemize}
        

\end{frame}

\subsection{Boundary Conditions}

\begin{frame}{Boundary Conditions}
    \begin{itemize}
        \item Key boundary conditions come from the payoff, given by: 
    \end{itemize} 
    \[
    payoff = strike - spot S(t)
    \]
    \begin{itemize}
        \item For a European call option: 
    \end{itemize}
    \[
    payoff = max[S(T)-K, 0]
    \]
    \begin{itemize}
        \item This payoff boundary condition becomes:
    \end{itemize}
    \[
    u(x,0) = max\left\{
    \exp\(\tfrac{1}{2}(k + 1) x\right)
    -
    \exp\(\tfrac{1}{2}(k - 1) x\right),
    0}
    \]
    \begin{itemize}
        \item Where
    \end{itemize}
    \[
    k = \frac{2 r}{\sigma^2}
    \]
    \[
    x = \ln\left(\frac{S(t)}{c(t)}\right)
    \]
    \[
    \tau = \frac{1}{2} \sigma^2 (T - t)
    \]
    \begin{itemize}
        \item Take log of price as stock price follows geometric Brownian motion
        \item Define rescale of time scaled by volatility
    \end{itemize}
\end{frame}

\subsection{Transformation to the Heat Equations}

\begin{frame}{Transformation to the Heat Equations}
    \begin{itemize}
        \item Define mapping from $c(s,t)$ to $f(x,\tau)$ with strike price $\K$ as scale factor:
    \end{itemize}
    \[
    c(s,t) = K f(x,\tau)
    \]
    \begin{itemize}
        \item The change of variables and mappings so far allow us to write the original PDE in another form:
    \end{itemize}
    \[
    \frac{\partial f}{\partial \tau} 
    =
    \frac{\partial^2 f}{\partial x^2}
    +
    (k - 1) \frac{\partial f}{\partial x}
    -
    k f
    \]
    \begin{itemize}
        \item We see this is a PDE with similar form to the heat equation, but with extra terms $(k - 1) \frac{\partial f}{\partial x}$ and $- k f$
        \item To make the extra terms disappear, we define a new mapping, where the exponential removes drift:
    \end{itemize}
    \[
    f(x,\tau) = \exp(\alpha x + \beta \tau) u(x,\tau)
    \]
    \begin{itemize}
        \item We then solve and choose $\alpha$ and $\beta$ such that the extra terms are removed, so we find:
    \end{itemize}
    \[
    \alpha = \frac{-1}{2} (k - 1)
    \]
    \[
    \beta = \frac{-1}{4} (k + 1)^2
    \]
    \begin{itemize}
        \item Which leads to a direct relationship between actual price of the option $c(s,t)$, and the modified price $u(x,\tau)$
        \item Substituting $\alpha$ and $\beta$ we find this relationship is given by:
    \end{itemize}
    \[
    c(s,t) = K u(x,\tau) \exp\left(\frac{-1}{2} (k - 1) x - \frac{1}{4} (k + 1)^2 \ tau\right)
    \]
    \begin{itemize}
        \item This allows us to conclude that the Black-Scholes equations can be expressed as the heat equation, where $\x$ is price of stock after transformations, $\tau$ is unit of time after transformations, and $u$ is the price of the option after transformations:
    \end{itemize}
    \[
    \frac{\partial u}{\partial \tau} 
    =
    \frac{\partial^2 u}{\partial x^2}
    \]
    \[
    - \infty < x < \infty
    \]
    \[
    \tau > 0
    \]
\end{frame}

\subsection{Final Solution}

\begin{frame}{Final Solution}
    \begin{itemize}
        \item We can then obtain the solutions to the Black-Scholes equation, where we have the price for European call and put options:
    \end{itemize}
    \[ 
    c = S_0 N(d_1) - K
    \exp(-r T) N(d_2)
    \]
    \[
    p = K
    \exp(-r T) N(-d_2) - S_0 N(-d_1)
    \]
    \[
    d_1 = \frac{\ln\left(\frac{S_0}{K}\right) + \left(r + \frac{\simga^2}{2}\right) (T - t)}{\sigma\sqrt{T - t}}
    \]
    \[
    d_2 = \frac{\ln\left(\frac{S_0}{K}\right) + \left(r - \frac{\sigma^2}{2}\right) (T - t)}{\sigma\sqrt{T - t}}
    =
    d_1 - \sigma\sqrt{T - t}
    \]
\end{frame}
