\section{Black-Scholes Equation}

\subsection{What is an Option?}

\begin{frame}{What is an Option?}
    \begin{itemize}
        \item Derivatives are financial contracts which derive their value from an underlying asset
        \item Futures, Forwards, Swaps, Options
        \item Options:
        \begin{itemize}
            \item Gives the right, not the obligation, to buy/sell asset at specified price on specified date
            \item Call gives right to buy
            \item Put gives right to sell
            \item American style can be utilised at any time up until expiration date
            \item European style can only be utilised on expiration date
        \end{itemize}
    \end{itemize}
\end{frame}

\subsection{Black-Scholes Equation}

\begin{frame}{Black-Scholes Background}
    \begin{itemize}
        \item Developed in 1970s by Fischer Black, Myron Scholes, and Robert Merton
        \item Provides an analytic solution to a second order parabolic partial differential equation by a transformation to the heat equation
        \item Leads to conclusion that options pricing dynamic is similar to heat dissipation
        \item Based on European style options
        \item American style options pricing can only be done numerically, not analytically
    \end{itemize}
\end{frame}

\begin{frame}{Black-Scholes Equation}
    The Black-Scholes equation is a second-order parabolic PDE. For a stock which does not pay dividends, we have:
    \[\frac{\partial c}{\partial t} + r S \frac{\partial c}{\partial S} + \frac{1}{2} \sigma^2 S^2 \frac{\partial^2 c}{\partial S^2} = rc\]
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item $t$ = time
                \item $c$ = call option price
                \item $p$ = put option price
                \item $S$ = spot stock price
                \item $r$ = risk free interest rate
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item $\sigma$ = stock volatility
                \item $S_0$ = spot stock price at time zero
                \item $K$ = strike price
                \item $T$ = time to maturity
            \end{itemize}
        \end{column}
    \end{columns}
    
    \vspace{10pt}
    Solutions to this PDE are the formulas for European call and put options prices.
\end{frame}

\begin{frame}{Boundary Conditions}
    \begin{itemize}
        \item Boundary conditions come from the payoff, given by:
        \[\text{payoff} = \text{strike} - \text{spot } S_t\]
        \item For a European call option, we have the terminal condition:
        \[\text{payoff} = \max\left\{S_T - K,\, 0\right\}\]
        \item So we have the lower bound of zero for $S = 0$, and the upper bound of $S$ as $S \to \infty$
    \end{itemize}
\end{frame}

\subsection{Transformation to the Heat Equation}

\begin{frame}{Transformation to the Heat Equation}
    Firstly, we define:
    \[x = \ln\!\left(\frac{S(t)}{K}\right) \qquad \text{and} \qquad \tau = \frac{1}{2} \sigma^2 (T - t)\]
    Take log of price as stock price follows geometric Brownian motion. \\
    Define rescale of time as time scaled by volatility. \\
    Define mapping from $c(s,t)$ to $f(x,\tau)$ with strike price $K$ as scale factor:
    \[c(s,t) = K f(x,\tau)\]
\end{frame}

\begin{frame}
    Substituting in the partial derivatives from Black-Scholes allows us to find the PDE:
    \[\frac{\partial f}{\partial \tau} = \frac{\partial^2 f}{\partial x^2} + \left(\frac{2r}{\sigma^2} - 1\right) \frac{\partial f}{\partial x} - \frac{2r}{\sigma^2} f\]
    which can be re-written as:
    \[\frac{\partial f}{\partial \tau} = \frac{\partial^2 f}{\partial x^2} + (k - 1) \frac{\partial f}{\partial x} - k f\]
    where $k = \dfrac{2r}{\sigma^2}$. We see this is a PDE with similar form to the heat equation, but with extra terms $(k - 1)\,\dfrac{\partial f}{\partial x}$ and $-kf$.
\end{frame}

\begin{frame}
    To make the extra terms disappear, we define a new mapping, where the exponential removes drift:
    \[f(x,\tau) = \exp(\alpha x + \beta \tau)\, u(x,\tau)\]
    We then solve and choose $\alpha$ and $\beta$ such that the extra terms are removed, so we find:
    \[\alpha = -\tfrac{1}{2}(k - 1) \qquad \text{and} \qquad \beta = -\tfrac{1}{4}(k + 1)^2\]
    which leads to a direct relationship between actual price of the option $c(s,t)$, and the modified price $u(x,\tau)$.
    Substituting $\alpha$ and $\beta$ we find this relationship is given by:
    \[c(s,t) = K\, u(x,\tau)\, \exp\!\left(-\tfrac{1}{2}(k - 1)\,x - \tfrac{1}{4}(k + 1)^2\,\tau \right)\]
\end{frame}

\begin{frame}
    This allows us to conclude that the Black-Scholes equation can be expressed as the heat equation, where $x$ is the price of the stock after transformations, $\tau$ is the unit of time after transformations, and $u$ is the price of the option after transformations:
    \[\frac{\partial u}{\partial \tau} = \frac{\partial^2 u}{\partial x^2}, \qquad -\infty < x < \infty, \qquad \tau > 0\]
    Where we have the payoff boundary condition:
    \[u(x,0) = \max\!\left\{\exp\!\left(\tfrac{1}{2}(k+1)\,x\right) - \exp\!\left(\tfrac{1}{2}(k-1)\,x\right),\; 0\right\}\]
\end{frame}

\begin{frame}{Final Solution}
    We can then obtain the solutions to the Black-Scholes equation, which shows the pricing of European call and put options:
    \[c = S_0 N(d_1) - K\exp(-rT)\,N(d_2) \qquad p = K\exp(-rT)\,N(-d_2) - S_0 N(-d_1)\]
    where
    \[d_1 = \frac{\ln\!\left(\dfrac{S_0}{K}\right) + \left(r + \dfrac{\sigma^2}{2}\right)(T - t)}{\sigma\sqrt{T - t}} \qquad \text{and} \qquad d_2 = d_1 - \sigma\sqrt{T - t}\]
\end{frame}
