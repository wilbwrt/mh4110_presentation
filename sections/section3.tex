\section{Experimentation and Results}

\begin{frame}{Neural Network Architecture}
\begin{itemize}
    \item \textbf{Structure:} 2-32-32-1
    \begin{itemize}
        \item 2 inputs
        \item 2 hidden layers with 32 neurons each
        \item 1 output (option price)
    \end{itemize} \pause
    
    \item \textbf{Activation Function:} Hyperbolic Tangent \pause
    
    \item \textbf{Training Algorithm:} Adam optimizer \pause
    
    \item \textbf{Training Duration:} 30,000 epochs
\end{itemize}
\end{frame}

% =========================
\begin{frame}{Computational Framework}
\begin{itemize}
    \item Implemented in \textbf{Python 3} \pause
    
    \item Built using the \textbf{Torch framework} \pause
    
    \item Utilized the \textbf{Neurodiffeq} library \pause
    
    \item Purpose: Solve differential equations using neural networks
\end{itemize}
\end{frame}

\begin{frame}{Mathematical Methodology}
\begin{itemize}
    \item Transformed the \textbf{Black-Scholes equation} into the heat equation \pause
    
    \item Change of variables:
    \begin{itemize}
        \item Generalized price: $x$
        \item Generalized time: $\tau$
    \end{itemize} \pause
    
    \item Applied real market payoff data as boundary conditions
\end{itemize}
\end{frame}
