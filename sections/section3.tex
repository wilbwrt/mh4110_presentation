    \begin{column}{0.50\textwidth}
      \textbf{Market:} Brazilian Stock Exchange (B3) \cite{santos2024}

      \vspace{0.5em}
      \begin{tabular}{lll}
        \toprule
        \textbf{Stock} & \textbf{Ticker} & \textbf{Series} \\
        \midrule
        Petrobras & PETR4 & PETRA (12 options) \\
                  &       & PETRD (11 options) \\
        Vale      & VALE3 & VALED (10 options) \\
        \bottomrule
      \end{tabular}

      \vspace{0.6em}
      \begin{itemize}
        \item Series A: maturity \textbf{January 2023}
        \item Series D: maturity \textbf{April 2023}
        \item \textbf{Call options only} (puts excluded due to low volume)
        \item Interest rate: SELIC = \textbf{13.75\%} p.a.\ (fixed)
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{Why Petrobras and Vale?}
        Highest trading volume on B3, giving dense and reliable time series. Other stocks such as Banco do Brasil and WEG had too many days without trades to be usable.
      \end{block}
      \vspace{0.5em}
      \begin{block}{Data transformation}
        Raw market prices are converted to $(x, \tau)$ coordinates before training. The payoff function serves as the initial boundary condition in transformed space.
      \end{block}
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{Error Metrics \cite{ferreira2008}}
  \vspace{0.2em}
  \footnotesize
  \renewcommand{\arraystretch}{1.4}
  \setlength{\tabcolsep}{5pt}
  \begin{tabular}{>{\bfseries}p{1.1cm} p{6.2cm} p{4.6cm}}
    \toprule
    \textbf{Metric} & \textbf{Formula} & \textbf{What it measures} \\
    \midrule
    MAE
      & $\displaystyle\frac{1}{N}\sum_{i=1}^{N} \bigl|Y_i - \hat{Y}_i\bigr|$
      & Average absolute error in BRL. Lower is better. \\[3pt]
    MSE
      & $\displaystyle\frac{1}{N}\sum_{i=1}^{N} (Y_i - \hat{Y}_i)^2$
      & Squares errors; penalises large misses more heavily. \\[3pt]
    MAPE
      & $\displaystyle\frac{1}{N}\sum_{i=1}^{N} \left|\frac{Y_i - \hat{Y}_i}{Y_i}\right|$
      & Relative error; comparable across different price levels. \\[3pt]
    POCID
      & $\displaystyle 100 \cdot \frac{\sum_{i=1}^{N} D_i}{N}$, \quad
        $D_i = \mathbf{1}_{\left[(Y_i\!-\!Y_{i-1})(\hat{Y}_i\!-\!\hat{Y}_{i-1}) > 0\right]}$
      & Fraction of days the model correctly predicted price direction. \\[3pt]
    ARV
      & $\displaystyle\frac{\sum_{i=1}^{N}(\hat{Y}_i - Y_i)^2}{\sum_{i=1}^{N}(\hat{Y}_i - \bar{Y})^2}$
      & Below 1 beats a naive mean predictor; 0 is a perfect model. \\
    \bottomrule
  \end{tabular}
\end{frame}

    \begin{column}{0.50\textwidth}
      \small
      \textbf{Graph color key \cite{santos2024}:}
      \begin{itemize}
        \item \textcolor{blue}{\textbf{Blue}} — actual option market price
        \item \textcolor{purple}{\textbf{Purple}} — underlying stock price (SPOT)
        \item \textcolor{magenta}{\textbf{Magenta}} — analytical B-S solution (BLS)
        \item \textcolor{nngreen}{\textbf{Green}} — neural network prediction (NN)
      \end{itemize}

      \vspace{0.25em}
      \textbf{What to look for:}
      Green tracks blue throughout. Magenta follows purple and only converges to blue near expiration.
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{exampleblock}{Benchmark \cite{liang2009}}
        Hong Kong options forecasting with NNs and SVR.\\[0.2em]
        Best MAE reported: \textbf{14.3}\\[0.2em]
        This paper (Petrobras): MAE $\approx$ \textbf{2.6}\\
        This paper (Vale): avg.\ MAE $\approx$ \textbf{5.6}
      \end{exampleblock}
    \end{column}
  \end{columns}
