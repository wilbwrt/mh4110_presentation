\section{Experimentation and Results}

\begin{frame}{Neural Network Architecture}
\begin{itemize}
    \item \textbf{Structure:} 2-32-32-1
    \begin{itemize}
        \item 2 inputs
        \item 2 hidden layers with 32 neurons each
        \item 1 output (option price)
    \end{itemize} \pause
    
    \item \textbf{Activation Function:} Hyperbolic Tangent \pause
    
    \item \textbf{Training Algorithm:} Adam optimizer \pause
    
    \item \textbf{Training Duration:} 30,000 epochs
\end{itemize}
\end{frame}

% =========================
\begin{frame}{Computational Framework}
\begin{itemize}
    \item Implemented in \textbf{Python 3} \pause
    
    \item Built using the \textbf{Torch framework} \pause
    
    \item Utilized the \textbf{Neurodiffeq} library \pause
    
    \item Purpose: Solve differential equations using neural networks
\end{itemize}
\end{frame}

% =========================
\begin{frame}{Mathematical Methodology}
\begin{itemize}
    \item Transformed the \textbf{Black-Scholes equation} into the heat equation \pause
    
    \item Change of variables:
    \begin{itemize}
        \item Generalized price: $x$
        \item Generalized time: $\tau$
    \end{itemize} \pause
    
    \item Applied real market payoff data as boundary conditions
\end{itemize}
\end{frame}

% =========================
\begin{frame}{Evaluation Metrics I}
To evaluate model performance:
\begin{itemize}
    \item \textbf{MAE} -- Mean Absolute Error \pause
    \begin{center}
        \begin{math}
            MAE = \frac{1}{N} \sum_{i=1}^{N} \left| Y_i - \hat{Y}_i \right|
        \end{math}
    \end{center}
    \item \textbf{MSE} -- Mean Square Error \pause
    \begin{center}
        \begin{math}
             MSE = \frac{1}{N} \sum_{i=1}^{N} ( Y_i - \hat{Y}_i )^2
        \end{math}
    \end{center}
    \item \textbf{MAPE} -- Mean Absolute Percentage Error \pause
    \begin{center}
        \begin{math}
            MAPE = \frac{1}{N} \sum_{i=1}^{N} \left| \frac{Y_i - \hat{Y}_i}{Y_i} \right|
        \end{math}
    \end{center}
\end{itemize}
\end{frame}

% =========================
\begin{frame}{Evaluation Metrics II}
To evaluate model performance:
\begin{itemize}
    \item \textbf{POCID} -- Prediction of Change in Direction \pause
    \begin{center}
        \begin{math}
            POCID = 100 \cdot \frac{\sum_{i=1}^{N} D_i}{N}
        \end{math}
    \end{center}
    where
    \begin{center}
        \begin{math}
        D_i =
        \begin{cases}
        1, & \text{if } (Y_i - Y_{i-1})(\hat{Y}_i - \hat{Y}_{i-1}) > 0 \\
        0, & \text{otherwise}
        \end{cases}
        \end{math}
    \end{center}
    \item \textbf{ARV} -- Average Relative Variance
    \begin{center}
        \begin{math}
        ARV = \frac{ \frac{1}{N} \sum_{i=1}^{N} (\hat{Y}_i - Y_i)^2 }
        { \frac{1}{N} \sum_{i=1}^{N} (\hat{Y}_i - \bar{Y})^2 }
        \end{math}
    \end{center}
\end{itemize}
\end{frame}

% =========================
\begin{frame}{Results}
    The experimental results demonstrate that the Multi-Layer Perceptron (MLP) neural network successfully learned to solve the modified Black-Scholes equation, providing more accurate option pricing forecasts than traditional analytical solutions.
\end{frame}
